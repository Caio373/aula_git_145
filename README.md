# Como fazer o primeiro commit no seu repositorio no GitHub

# Passo 1
git add .

# Passo 2
git commit -m "first commit"

# Passo 3
git branch -M main

# Passo 4 
git remote add origin <URL_DO_REPOSITORIO>.git

# Passo 5
git push -u origin main
